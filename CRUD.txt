    AGGREGATION

        # Operators
            $match      $group      $project        $skip
            $sort       $count      $limit          $out

        # $group
            db.collection_name.aggregate({$group: {_id: "$age"}})
            db.collection_name.aggregate({$group: {_id: "$name", {$sum: "$price"}}})

        # $match
            aggregate({$match: {age: {$gt: 24}}})
            aggregate({$match: {$and:[
                {gender: "female"},
                {age: {$gt: 30 }}
